Mini-Lab1​

思路​

Linux下的进程信息保存在/proc下，打开/proc查看文件内容

/proc/stat保存的是cpu等一系列信息，可访问的数字文件夹表示的是当前进程pid

查看某一进程文件夹，比如进程号为82的进程，用cat stat | less查看文件夹中stat中的信息

/proc/[pid]/stat中的信息组成如下：

进程号（pid） 进程名（comm） 进程状态（state） 父进程（p_pid）

用Linux自带的pstree命令查看一下当前的进程

可以看到pid=82的进程确实是(init)

有了进程号和父进程号就可以用递归输出进程树了

Linux的文件和文件夹操作可以用dirent

dirent是POSIX.1标准定义的unix类目录操作的头文件，结构体定义如下

// 获取/proc/下所有进程id并返回进程数量
int getProcNumber() {
    DIR *dir = opendir(PROC);
    struct dirent *file;

    // 如果无进程或访问失败
    if (dir == NULL) {
        return -1;
    }

    int cnt = 0; // 进程数量

    // 遍历"/proc/"下所有的文件，只包含数字的就是进程文件
    while ((file = readdir(dir)) != NULL) {
        if (strncmp(file->d_name, ".", 1) == 0) {
            continue;
        }
        int len = strlen(file->d_name);
        bool flag = TRUE;
        // 遍历文件名查看是否有非数字字符
        for (int i = 0; i < len; i++) {
            if(!isdigit(file->d_name[i])) {
                flag = FALSE;
                break;
            }
        }
        if(flag == FALSE) {
            continue;
        }
        strcpy(proc_ids[cnt++], file->d_name);
    }

    closedir(dir);

    return cnt;
}


// 通过id获取进程信息，状态在/proc/[id]/stat下
int getProcess(char *id, Process *out) {
    static char stat_path[512];
    static char buf[1024];
    static char tmp[128];

    sprintf(stat_path, "%s/%s/stat", PROC, id);

    FILE *stat = fopen(stat_path, "rt");

    if (stat == NULL) {
        return -1;
    }

    fgets(buf, sizeof(buf), stat);

    fclose(stat);

    sscanf(buf, "%d %s %c %d", &out->pid, tmp, &out->state, &out->p_pid);

    // 处理进程名，buf读入的进程名是包括左右括号的
    strcpy(out->comm, tmp + 1);
    out->comm[strlen(out->comm) - 1] = '\0';

    return 0;
}

// 根据进程名排序
int cmpByName(const void *a, const void *b) {
    int i = *((int *)a);
    int j = *((int *)b);
    return strcmp(procs[i].comm, procs[j].comm);
}

// 根据进程pid排序
int cmpByPid(const void *a, const void *b) {
    int i = *((int *)a);
    int j = *((int *)b);
    if (procs[i].pid > procs[j].pid) {
        return 1;
    }
    else if(procs[i].pid < procs[j].pid) {
        return -1;
    }
    else {
        return 0;
    }
}

void showTree(int rid) {
    static int offset = 0; // 静态offset
    static char tmp[1024];

    int children[MAX], ch_cnt = 0;

    Process *root = &procs[rid];

    for (int i = 0; i < offset; i++) {
        putchar(' ');
    }

    if (cli.show_pids) {
        sprintf(tmp, "%s(%d)", root->comm, root->pid);
    } else {
        sprintf(tmp, "%s", root->comm);
    }

    puts(tmp);

    showd[rid] = TRUE;

    int delta = 4;
    offset += delta;

    for (int i = 0; i < proc_number; i++) {
        if (procs[i].pid == -1) {
            continue;
        }
        if (procs[i].p_pid != root->pid) {
            continue;
        }
        children[ch_cnt++] = i;
    }

    if (cli.numberic_sort) {
        qsort(children, ch_cnt, sizeof(int), cmpByName);
    } else {
        qsort(children, ch_cnt, sizeof(int), cmpByPid);
    }

    // 递归输出进程数
    for (int i = 0; i < ch_cnt; i++) {
        showTree(children[i]);
    }

    offset -= delta;
}

int main(int argc, char *argv[]) {
    cli.show_pids = FALSE;
    cli.version = FALSE;
    cli.numberic_sort = FALSE;

    // 解析命令
    for (int i = 0; i < argc; i++) {
        // 保证有命令输入
        assert(argv[i]);

        if (strcmp(argv[i], "-p") == 0 || strcmp(argv[i], "--show-pids") == 0) {
            cli.show_pids = TRUE;
        }

        if (strcmp(argv[i], "-V") == 0 || strcmp(argv[i], "--version") == 0) {
            cli.version = TRUE;
        }

        if (strcmp(argv[i], "-n") == 0 || strcmp(argv[i], "--numberic-sort") == 0) {
            cli.numberic_sort = TRUE;
        }
    }
    // C标准保证
    assert(!argv[argc]);
    
    // 打印版本号
    if (cli.version) {
        printf("pstree1.0\nCopyright (C) 2022-2022 TGFpbg64\n");
        return EXIT_SUCCESS;
    }

    proc_number = getProcNumber();

    for (int i = 0; i < proc_number; i++) {
        int judge = getProcess(proc_ids[i], procs + i);
        if (judge != 0) {
            procs[i].pid = -1;
        }
    }

    for (int i = 0; i < proc_number; i++) {
        showd[i] = FALSE;
    }

    for (int i = 0; i < proc_number; i++) {
        if (procs[i].p_pid == 0) {
            showTree(i);
        }
    }
}

Mini-Lab2​

libco要实现的就是co_start、co_wait、co_yield三个函数

首先定义协程

typedef enum {
  CO_FREE, // 闲置
  CO_NEW, // 新创建，还未执行过，准备调度
  CO_RUNNING, // 已经执行过
  CO_WAITING, // 在 co_wait 上等待
  CO_SUSPEND, // 挂起
  CO_DEAD, // 已经结束，但还未释放资源
} co_status;

struct co {
  int id;
  char *name;
  void (*func)(void *); // co_start 指定的入口地址和参数
  void *arg;

  co_status status; // 协程的状态
  struct co *    waiter; // 是否又其他协程在等待当前协程
  ucontext_t     ctx, pctx;// 寄存器跳转存储
  uint8_t        stack[STACK_SIZE]; // 协程的堆栈
};

在struct co中，使用了ucontext函数族，是用来帮助用户更方便的实现coroutine，类似setjmp/longjmp

ucontext函数有4个

#include <ucontext.h>

// 用户上下文的获取和设置
int getcontext(ucontext_t *ucp);
int setcontext(const ucontext_t *ucp);

// 操纵用户上下文
void makecontext(ucontext_t *ucp, void (*func)(void), int argc, ...);
int swapcontext(ucontext_t *oucp, const ucontext_t *ucp);

4 个函数都依赖于ucontext_t类型，这个类型大致为：

typedef struct {
 ucontext_t *uc_link;
 sigset_t    uc_sigmask;
 stack_t     uc_stack;
 mcontext_t  uc_mcontext;
 ...
} ucontext_t;

其中：

  • uc_link：当前上下文结束时要恢复到的上下文，其中上下文由makecontext()创建；
  • uc_sigmask：上下文要阻塞的信号集合；
  • uc_stack：上下文所使用的 stack；
  • uc_mcontext：其中 mcontext_t

4个函数的作用分别是：

  • getcontext()：将当前的 context 保存在ucp中。
  • setcontext()：恢复用户上下文为ucp所指向的上下文，成功调用不用返回。如果上下文是由 getcontext() 产生，则切换到该上下文后，程序的执行在getcontext()后继续执行。如果上下文是由 makecontext() 产生，切换到该上下文，程序的执行切换到makecontext()调用所指定的第二个参数的函数上。当函数返回后，如果ucp.uc_link为 NULL，则结束运行；反之跳转到对应的上下文。
  • makecontext()：修改ucp所指向的上下文；
  • swapcontext()：保存当前的上下文到ocup，并且设置到ucp所指向的上下文。

创建协程​

/* 创建一个协程 */
static int creat_co(const char *name, void (*func)(void *), void *arg) {
  int id = get_co(CO_FREE, rand() % MAX_CO);
  if(id == -1) {
    return -1;
  }

  struct co *thd = coroutines_pool + id;
  memset(thd->stack, 0, STACK_SIZE);
  thd->status = CO_NEW;
  thd->name = (char *)name;
  thd->func = (*func);
  thd->arg = arg;
  
  getcontext(&thd->ctx);
  thd->ctx.uc_stack.ss_sp = thd->stack;
  thd->ctx.uc_stack.ss_size = STACK_SIZE;
  thd->ctx.uc_stack.ss_flags = 0;
  thd->ctx.uc_link = &thd->pctx;
  makecontext(&thd->ctx, (void (*)(void))(co_func_wrap), 1, thd);

  return id;
}

通过get_co()随机获得一个协程池里的协程，对创建出来的新协程做好初始化，用getcontext()保存上下文

必须先用getcontext保存，因为在getcontext中保存了浮点寄存器相关值，该值在swapcontext、setcontext中都有使用，否则调用时会发生异常。

用makecontext()封装好协程就完成了创建

struct co *coroutines_pool;
static int current_co;

/* 初始化协程池 */
void init_co() {
  coroutines_pool = malloc(MAX_CO * sizeof(struct co));
  current_co = -1;
  for(int i = 0; i < MAX_CO; i++) {
    coroutines_pool[i].id = i;
    coroutines_pool[i].arg = NULL;
    coroutines_pool[i].status = CO_FREE;
    memset(coroutines_pool[i].stack, 0, STACK_SIZE);
  }
}

/* 封装一个协程 */
void co_func_wrap(struct co *thd) {
  assert(thd != NULL);
  thd->func(thd->arg);
  thd->status = CO_FREE;
  current_co = -1;
}

/* 随机获取一个协程 */
static int get_co(co_status status, int num) {
  assert(num >= 0 && num < MAX_CO);
  for(int i = 0; i < MAX_CO; i++) {
    int id = (num + i) % MAX_CO;
    if(coroutines_pool[id].status == status) {
      return id;
    }
  }
  return -1;
}

管道(pipe)​

管道用于进程之间的通信，一般是父子进程。

用int pipe(int fd[2])创建管道，管道的两端用描述符fd[0]和fd[1]描述，描述字fd[0]表示读，fd[1]表示写

使用dup2重定向管道的两端描述符，比如dup2(fd[0], STDIN_FILENO)就是将fd[0]读端重定向到了标准输入，这样就可以通过访问fd[0]来获取标准输入

typedef struct {

syscallItem syscalls[MAX_SYSCALL];

static time_t last_time, now_time;

int Judge_String(const char *str) {
  int length = strlen(str);
  if (length > 0 && isalpha(str[0]) && str[length - 1] == '>') {
    return 1;
  }
  else {
    return 0;
  }
}

void AddItem(char *name, double run_time) {
  for (int i = 0; i < total_call; i++) {
    if(strcmp(name, syscalls[i].name) == 0) {
      syscalls[i].total_time += run_time;
      return ;
    }
  }
  strcpy(syscalls[total_call].name, name);
  syscalls[total_call].total_time = run_time;
  total_call++;
}

void GetData(const char *str, char *name, double *run_time) {
  int length = strlen(str);
  static char time_buffer[MAX_LENGTH];
  // 系统调用名
  for (int i = 0; i < length; i++) {
    if (str[i] == '(') {
      name[i] = '\0';
      break;
    }
    name[i] = str[i];
  }
  // 系统调用时间
  int buf_pos = 0;
  for (int i = length - 9; i < length; i++) {
    if (str[i] == '>') {
      time_buffer[buf_pos++] = '\0';
      break;
    }
    time_buffer[buf_pos++] = str[i];
  }
  *run_time = atof(time_buffer);
}

int cmp(const void *a, const void *b) {
  syscallItem *ta = (syscallItem *)a;
  syscallItem *tb = (syscallItem *)b;
  if (ta->total_time > tb->total_time) {
    return -1;
  }
  else if (ta->total_time < tb->total_time) {
    return 1;
  }
  else {
    return strcmp(ta->name, tb->name);
  }
}

void Print_Screen() {
  static syscallItem temp_syscalls[MAX_SYSCALL];
  int temp_totcall;
  double sum_time = 0.0;

  // 信息快照
  temp_totcall = total_call;
  memcpy(temp_syscalls, syscalls, sizeof(syscallItem) * total_call);

  for (int i = 0; i < temp_totcall; i++) {
    sum_time += temp_syscalls[i].total_time;
  }
  
  qsort(temp_syscalls, temp_totcall, sizeof(syscallItem), cmp);

  system("clear");

  for (int i = 0; i < temp_totcall; i++) {
    printf("%20s: %10.5lf%%\n", temp_syscalls[i].name, temp_syscalls[i].total_time / sum_time * 100);
  }
  printf("\n");
  time(&last_time);
}

int Parent_Proc(int pid) {
  static char string_buffer[MAX_LENGTH];
  static char name_buffer[MAX_LENGTH];
  static double run_time;

  time(&last_time);

  total_call = 0;

  while(scanf("%[^\n]\n", string_buffer) == 1) {
    int length = strlen(string_buffer);
    if (Judge_String(string_buffer)) {
      GetData(string_buffer, name_buffer, &run_time);
      AddItem(name_buffer, run_time);
    }
    time(&now_time);
    if(now_time - last_time > 0.001)  {
      Print_Screen();
    }
  }
#ifdef Debug

  double sum_time = 0.0;

  for (int i = 0; i < total_call; i++) {
    sum_time += syscalls[i].total_time;
  }

  for (int i = 0; i < total_call; i++) {
    printf("syscall_name: %s | time_percent: %lf\n", syscalls[i].name, syscalls[i].total_time / sum_time * 100);
  }

#endif
  Print_Screen();
  return 1;
}

void Child_Proc(int argc, char *argv[]) {
  char *cargv[argc + 3];
  cargv[0] = "strace";
  cargv[1] = "-T";
  for (int i = 0; i < argc; i++) {
    cargv[i + 2] = argv[i];
  }
  cargv[argc + 2] = NULL;
  execvp("strace", cargv);
}

    dup2(open("/dev/null", O_RDWR), STDOUT_FILENO);

    close(fd[0]);

    Child_Proc(argc - 1, argv + 1);
  }
  else {

    close(fd[0]);
    close(fd[1]);
    return Parent_Proc(pid);
  }
}

Mini-Lab4​

思路​

  1. 获取一行输入，判断是函数还是表达式，分开处理
  2. 如果前几个字符是int则是函数，否则就是表达式
  3. 如果是函数，则需要把输入的函数输出到一个.so的动态链接库中，方便后面调用
  4. 如果是表达式，可以用一个函数封装表达式，只需要在返回值里输出表达式即可

需要处理的点​

  • static变量和普通的变量要区分开，根据数据的使用方式决定使用哪种变量。
  • 动态链接的时候使用的是#include <dlfcn.h>这个库

dlfcn.h中主要包括四个函数

#include <dlfcn.h>
void* dlopen(const char*, int flag);
char* dlerror();
void* dlsym(void* handler, char* symbol);
int dlclose(void* handler);

dlopen是打开一个动态链接库，返回一个类型为void *的handle，flag有两个值

  • RTLD_LAZY暂缓决定，等有需要时再解出符号
  • RTLD_NOW立即决定，返回前解除所有未决定的符号

dlsym是查找动态链接库中的符号symbol（函数）并返回符号地址

FILE *CreatLinkso(char **so_name, char **raw_so_name) {
  const char temp_so_name[MAX_BUFFER] = "/tmp/crepl_XXXXXX";
  static char _so_name[MAX_BUFFER], _raw_so_name[MAX_BUFFER * 2];

  strcpy(_raw_so_name, temp_so_name);

  if (mkstemp(_raw_so_name) == -1) {
    return NULL;
  }

  strcpy(_so_name, _raw_so_name + 5);

  FILE *file = fopen(_raw_so_name, "w");
  assert(file != NULL);

  for (FUNC_NODE *cur = head; cur != NULL; cur = cur->next) {
    fprintf(file, "%s\n", cur->content);
  }

  *so_name = _so_name;
  *raw_so_name = _raw_so_name;
  
  return file;
}

int Compile(char *so_name, char *raw_so_name, char **compileName) {
  static char buffer[MAX_BUFFER * 4];
  static char _compiled[MAX_BUFFER];
  
  sprintf(_compiled, "/tmp/%s.so", so_name);

#if defined(__i386__)
  sprintf(buffer, "gcc -x c %s -m32 -shared -fPIC -Wall -o %s", raw_so_name, _compiled);
#elif defined(__x86_64__)
  sprintf(buffer, "gcc -x c %s -shared -fPIC -Wall -o %s", raw_so_name, _compiled);
#endif

#ifdef Debug

  printf("%s\n", raw_so_name);

#endif

  int res = system(buffer);
  *compileName = _compiled;
  
  return res;
}

void AddItem() {
  FUNC_NODE *temp = malloc(sizeof(FUNC_NODE));
  if (head == NULL) {
    head = temp;
    tail = head;
    tail->content = malloc(sizeof(char) * (screen_input.length + 1));
    strcpy(tail->content, screen_input.content);
    tail->next = NULL;    
  }
  else {
    tail->next = temp;
    tail = temp;
    tail->content = malloc(sizeof(char) * (screen_input.length + 1));
    strcpy(tail->content, screen_input.content);
    tail->next = NULL;
  }
}

char *WrapFunc(FILE *file, char *expr, char *name) {
  static char _wrap_name[MAX_BUFFER * 2];
  sprintf(_wrap_name, "__expr_wrapper_%s", name);
  fprintf(file, "int %s() {\n    return (%s);\n}\n", _wrap_name, expr);
  return _wrap_name;
}

void DynamicLink() {
  char *so_name, *raw_so_name;
  char *compiled;
  char *wrap_name;

  FILE *file = CreatLinkso(&so_name, &raw_so_name);

  assert(file != NULL);

  if (screen_input.is_func) {
    fputs(screen_input.content, file);
    fclose(file);

#ifdef Debug

  printf("%s\n", raw_so_name);

#endif

    if (Compile(so_name, raw_so_name, &compiled) == 0) {
      AddItem();
      printf("Added: %s\n", screen_input.content);
    }
  }
  else {
    wrap_name = WrapFunc(file, screen_input.content, so_name);
    fclose(file);

    if (Compile(so_name, raw_so_name, &compiled) == 0) {
      // 打开动态链接库
      void *handle = dlopen(compiled, RTLD_NOW);
      assert(handle != NULL);

      int (*wrap_func)() = dlsym(handle, wrap_name);
      assert(wrap_func != NULL);

      printf("(%s) == %d\n", screen_input.content, wrap_func());
    }
  }
}
